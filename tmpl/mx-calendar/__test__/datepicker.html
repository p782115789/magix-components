mx-calendar

默认情形

HTML Code:
<input mx-view="app/gallerymx-calendar/datepicker" class="input" placeholder="请选择日期">

默认选中某天

HTML Code:
<input class="input" placeholder="请选择日期"
    mx-view="app/gallery/mx-calendar/datepicker"
    view-selected="2018-01-01">

从input中读取日期

HTML Code:
<input class="input" placeholder="请选择日期"
    value="2018-03-02"
    mx-view="app/gallery/mx-calendar/datepicker">

设置可选择的范围

HTML Code:
<input mx-view="app/gallery/mx-calendar/datepicker"
    view-min="2017-08-05"
    view-max="2017-08-22"
    value="2017-08-12"
    class="input"
>

设置周二为一周的开始

HTML Code:
<input mx-view="app/gallery/mx-calendar/datepicker"
    view-week-start="<%@ 2 %>"
    class="input"
>

change事件

HTML Code:
<input mx-view="app/gallery/mx-calendar/datepicker"
    view-time-types="all"
    view-selected="2018-08-08 18:08:20"
    mx-change="showDatetime()"
    class="input"
>
Javascript Code:
let Magix = require('magix');
let GTip = require('mx-gtip/index');
module.exports = Magix.View.extend({
    tmpl: '@index.html',
    mixins: [GTip],
    render() {
        let me = this;
        me.updater.digest();
    },
    'showDatetime<change>' (e) {
        this.gtipRT('日期：' + e.date + '，时间：' + e.time);
    }
});
