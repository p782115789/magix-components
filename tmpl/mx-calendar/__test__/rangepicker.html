mx-calendar

默认情形

HTML Code:
<input mx-view="app/gallery/mx-calendar/rangepicker" class="input" placeholder="请选择日期范围">

指定开始结束日期

HTML Code:
<input class="input" placeholder="请选择日期范围"
    mx-view="app/gallery/mx-calendar/rangepicker"
    view-start="2018-01-01"
    view-end="2019-01-01">

不显示快捷日期

HTML Code:
<input class="input" placeholder="请选择日期范围"
    mx-view="app/gallery/mx-calendar/datepicker"
    view-shortcuts="false">

设置可选择的范围

HTML Code:
<input mx-view="app/gallery/mx-calendar/rangepicker"
    view-min="2017-08-05"
    view-max="2017-08-22"
    view-start="2017-08-08"
    view-end="2017-08-20"
    view-shortcuts="false"
    class="input"
>

change事件

HTML Code:
<input mx-view="app/gallery/mx-calendar/rangepicker"
    view-time-types="all"
    mx-change="showDatetime()"
    class="input"
>
Javascript Code:
let Magix = require('magix');
let GTip = require('mx-gtip/index');
module.exports = Magix.View.extend({
    tmpl: '@rangepicker.html',
    mixins: [GTip],
    render() {
        let me = this;
        me.updater.digest();
    },
    'showDatetime<change>' (e) {
        console.log(e.dates);
        this.gtipRT('日期：' + JSON.stringify(e.dates));
    }
});
