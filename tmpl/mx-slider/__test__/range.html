响应change事件

HTML Code:
<div mx-view="mx-slider/range"
    view-max="500"
    view-min="200"
    view-value="210,350"
    view-step="10"
    mx-change="showValue()"
    class="fl"
></div>
Javascript Code:
let Magix = require('magix');
let GTip = require('app/gallery/mx-gtip/index');
module.exports = Magix.View.extend({
    tmpl: '@index.html',
    mixins: [GTip],
    render() {
        let me = this;
        me.updater.digest();
    },
    'showValue<change>' (e) {
        this.gtipRT('value:' + e.value);
    }
});
